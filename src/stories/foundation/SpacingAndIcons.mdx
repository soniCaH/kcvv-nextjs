## Usage in code

```tsx
// Named import from @/lib/icons
import { Trophy, MapPin } from "@/lib/icons";
<Trophy className="w-5 h-5 text-kcvv-green" />

// Via Icon component (design-system wrapper)
import { Icon } from "@/components/design-system/Icon";
<Icon name="trophy" size="md" color="primary" />
```
